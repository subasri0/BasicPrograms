package BasicPrograms;
import java.util.Scanner;
public class AlphabetTrianglePattern 
{
	public static void main(String[] args)
	{
		int n = 65;
		for(int i = 0 ; i < 5 ; i++)
		{
			for(int k=1;k<5-i;k++)
			{
				System.out.print(" ");
			}
			//for printing alphabets
			for(int j=0;j<=i;j++)
			{
				System.out.print((char)(n+j)+" ");
			}
			System.out.println();
		}
	}
}
